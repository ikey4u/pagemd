Removes unwanted elements before extraction. Modifies the live DOM. Returns { removed }.

// Remove header, footer, sidebar, ads
(function() {
  const sels = ['header', 'footer', 'nav', '.sidebar',
    '.ads', '[role="banner"]', '.cookie-notice'];
  let removed = 0;
  sels.forEach(s =>
    document.querySelectorAll(s).forEach(el => { el.remove(); removed++; })
  );
  return { removed };
})()

Extracts content. Returns { title, html } or null. Use cloneNode to avoid changing the page.

// Extract <article>, remove sidebar
(function() {
  const el = document.querySelector('article');
  if (!el) return null;
  const clone = el.cloneNode(true);
  clone.querySelectorAll('.sidebar').forEach(e => e.remove());
  return { title: document.title, html: clone.innerHTML };
})()

Navigates to next page. Returns { success }.

// Click link containing "下一页"
(function() {
  const btns = [...document.querySelectorAll('a, button')];
  const next = btns.find(b => b.textContent.includes('下一页'));
  if (!next) return { success: false };
  next.click();
  return { success: true };
})()

Stops batch crawling. Receives context (currentUrl, pageIndex, collectedUrls[], collectedTitles[]). Returns { shouldStop, reason? }.

// Stop when URL contains "example.com"
(function(context) {
  if (context.currentUrl.includes('example.com'))
    return { shouldStop: true, reason: 'Reached target' };
  return { shouldStop: false };
})()

// Stop on repeated URL (loop detection)
(function(context) {
  if (context.collectedUrls.includes(context.currentUrl))
    return { shouldStop: true, reason: 'Loop detected' };
  return { shouldStop: false };
})()
