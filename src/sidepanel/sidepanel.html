Clean optional

Removes unwanted elements before extraction. Modifies the live DOM.
Returns { removed: number }

(function() {
  const sels = ['header', 'footer', 'nav',
    '.sidebar', '.ads', '.cookie-notice'];
  let removed = 0;
  sels.forEach(s =>
    document.querySelectorAll(s).forEach(el => {
      el.remove(); removed++;
    })
  );
  return { removed };
})()

Extract

Extracts page content. Use cloneNode to avoid modifying the page.
Returns { title: string, html: string } or null

(function() {
  const el = document.querySelector('article');
  if (!el) return null;
  const clone = el.cloneNode(true);
  clone.querySelectorAll('.sidebar').forEach(
    e => e.remove()
  );
  return {
    title: document.title,
    html: clone.innerHTML
  };
})()

Navigate optional

Clicks next page or performs navigation.
Returns { success: boolean }

(function() {
  const btns = [...document.querySelectorAll('a, button')];
  const next = btns.find(
    b => b.textContent.includes('下一页')
  );
  if (!next) return { success: false };
  next.click();
  return { success: true };
})()

Stop optional

Decides when to stop batch crawling. Receives context with currentUrl, pageIndex, collectedUrls[], collectedTitles[].
Returns { shouldStop: boolean, reason?: string }

(function(context) {
  if (context.currentUrl.includes('example.com'))
    return { shouldStop: true, reason: 'Done' };
  if (context.collectedUrls.includes(
    context.currentUrl
  ))
    return { shouldStop: true, reason: 'Loop' };
  return { shouldStop: false };
})()
